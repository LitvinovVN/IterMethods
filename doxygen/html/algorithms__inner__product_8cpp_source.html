110 {
111 int NbyThread = n / NumCpuThreads;
112 //printf("NbyThread = %d\n", NbyThread);
113 for(size_t i = 0; i < NumCpuThreads; i++)

115 }
116 else
117 {
118 int NbyThread = n / (NumCpuThreads-1);

125 threads.push_back(std::thread(scalar_mult_cpu_thread, v1, v2, (NumCpuThreads-1)*NbyThread, NbyThreadLast, std::ref(res)));
126 }

131 thread.join();
